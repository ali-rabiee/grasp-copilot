\begin{figure}[t]
  \centering
  \includegraphics[width=\linewidth]{figs/fig1_per_env_bars.pdf}
  \caption{Tool-call accuracy by environment on held-out oracle validation
  sets. The headline model (Qwen2.5-3B-Oracle$\rightarrow$WoZ-LoRA) matches or
  exceeds Oracle-LoRA on every environment, and substantially outperforms
  rule-based and shared-autonomy baselines on the new Stacking and Pouring
  environments where the baselines lack a vocabulary for STACK/GRAB/POUR.}
  \label{fig:per_env_bars}
\end{figure}

\begin{figure}[t]
  \centering
  \includegraphics[width=\linewidth]{figs/fig2_ambiguous_gap.pdf}
  \caption{Tool-call accuracy on clean vs.\ ambiguous scenarios (macro-averaged
  across the three environments). WoZ supervision closes the gap that
  heuristic shared-autonomy baselines suffer on ambiguous scenes. Arrows
  indicate the absolute shift; trained LoRAs (left three bars) all reach or
  approach 100\% on the ambiguous set, while H1 and SA1 drop by 8 and 14
  percentage points respectively.}
  \label{fig:ambiguous_gap}
\end{figure}

\begin{figure}[t]
  \centering
  \includegraphics[width=\linewidth]{figs/fig4_confusion_grid.pdf}
  \caption{Row-normalized tool-call confusion matrices on the YCB held-out
  validation set. Trained LoRAs concentrate mass on the diagonal; heuristic
  baselines (H1 / H2) systematically over-emit \texttt{INTERACT}, sacrificing
  motion accuracy for conversational safety.}
  \label{fig:confusion_grid}
\end{figure}

\begin{figure}[t]
  \centering
  \includegraphics[width=\linewidth]{figs/fig5_context_heatmap.pdf}
  \caption{Tool-call accuracy by dialog-context type, per model. Trained LoRAs
  remain accurate across both \emph{cold-start} (no prior dialog) and
  \emph{follow-up} contexts (confirmation, candidate-choice, intent-gate),
  while heuristic baselines specialize on the cold-start case.}
  \label{fig:context_heatmap}
\end{figure}

\begin{figure}[t]
  \centering
  \includegraphics[width=\linewidth]{figs/robustness_curves.pdf}
  \caption{Tool-call accuracy under simulated teleoperation noise
  ($p \in \{0, 0.1, 0.2, 0.3, 0.5\}$ of gripper-history cells and yaw bins
  jittered to a random neighbor). Solid lines: trained LoRAs. Dashed: Oracle
  reference. Dotted: H1 (Ask-if-Ambiguous). The headline model
  Oracle$\rightarrow$WoZ-LoRA degrades the most gracefully, retaining the bulk
  of its advantage at high noise levels.}
  \label{fig:robustness}
\end{figure}
